# const
const x = 42;
==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,Name),Expr(TypeExpr(Integer)))))

# var
var x = 42;
==>
Program(Decl(GlobalVarDecl(VarDeclProto(var,Name),Expr(TypeExpr(Integer)))))

# const with type
const x: i32 = 42;
==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,Name,TypeExpr(Identifier)),Expr(TypeExpr(Integer)))))

# var with type
var x: f64 = 3.14;
==>
Program(Decl(GlobalVarDecl(VarDeclProto(var,Name,TypeExpr(Identifier)),Expr(TypeExpr(Float)))))

# multiline string
const s = \""""
==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,Name),⚠,Expr(TypeExpr(StringLiteral)),⚠)),ContainerField(TypeExpr(StringLiteral)))

# fn declaration
fn foo() void { }
==>
Program(Decl(FnProto(fn,Identifier,ParamDeclList("(",")"),TypeExpr(Identifier)),Block("{","}")))

# fn with return
fn add(a: i32, b: i32) i32 { return a + b; }
==>
Program(Decl(FnProto(fn,Identifier,ParamDeclList("(",ParamDecl(Identifier,ParamType(TypeExpr(Identifier))),ParamDecl(Identifier,ParamType(TypeExpr(Identifier))),")"),TypeExpr(Identifier)),Block("{",Statement(Expr(return,Expr(BinaryExpr(Expr(TypeExpr(Identifier)),AdditionOp,Expr(TypeExpr(Identifier)))))),"}")))

# pub fn
pub fn main() void { }
==>
Program(pub,Decl(FnProto(fn,Identifier,ParamDeclList("(",")"),TypeExpr(Identifier)),Block("{","}")))

# fn with params
fn process(x: i32, y: i32) void { }
==>
Program(Decl(FnProto(fn,Identifier,ParamDeclList("(",ParamDecl(Identifier,ParamType(TypeExpr(Identifier))),ParamDecl(Identifier,ParamType(TypeExpr(Identifier))),")"),TypeExpr(Identifier)),Block("{","}")))

# struct declaration
const Point = struct { x: i32, y: i32 };
==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,Name),Expr(TypeExpr(ContainerDecl(ContainerDeclType(struct),ContainerBlock("{",ContainerField(Identifier,TypeExpr(Identifier)),ContainerField(Identifier,TypeExpr(Identifier)),"}")))))))

# empty struct
const Empty = struct { };
==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,Name),Expr(TypeExpr(ContainerDecl(ContainerDeclType(struct),ContainerBlock("{","}")))))))

# enum declaration
const Color = enum { red, green, blue };
==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,Name),Expr(TypeExpr(ContainerDecl(ContainerDeclType(enum),ContainerBlock("{",ContainerField(TypeExpr(Identifier)),ContainerField(TypeExpr(Identifier)),ContainerField(TypeExpr(Identifier)),"}")))))))

# test declaration
test "addition" { try expect(add(1, 2) == 3); }
==>
Program(TestDecl(test,TestName(StringLiteralSingle),Block("{",Statement(Expr(PrefixExpr(PrefixOp(try),TypeExpr(Identifier,FnCallArgs("(",Expr(BinaryExpr(Expr(TypeExpr(Identifier,FnCallArgs("(",Expr(TypeExpr(Integer)),Expr(TypeExpr(Integer)),")"))),CompareOp,Expr(TypeExpr(Integer)))),")"))))),"}")))

# usingnamespace
usingnamespace std;
==>
Program(Decl(usingnamespace,Expr(TypeExpr(Identifier))))
